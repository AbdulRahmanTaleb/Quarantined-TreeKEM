     either {
       with(authService) {
         val cached = commitCache[makeCommitRef(handshakeMessage).hex]
 
         val newState =
           state.ensureActive {
             process(handshakeMessage, authService, psks = psks, cachedState = cached?.newState)

         when (newState.epoch) {

             }
         }
       }
     }
